# List Transactions

Retrieve a paginated list of all transactions for your account with optional filtering and sorting capabilities.

## Endpoint

<Badge variant="info">**GET**</Badge> `/transactions`

## Headers

| Name | Type | Required | Description |
|------|------|----------|-------------|
| `Authorization` | string | **Required** | Bearer token with your secret API key |
| `Content-Type` | string | **Required** | Must be `application/json` |

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination (starts from 1)
</ParamField>

<ParamField query="per_page" type="integer" default="50">
  Number of transactions per page (maximum 100)
</ParamField>

<ParamField query="status" type="string">
  Filter transactions by status
  
  **Available statuses:** `success`, `failed`, `pending`, `abandoned`, `cancelled`
</ParamField>

<ParamField query="currency" type="string">
  Filter transactions by currency code (e.g., `NGN`, `USD`, `GBP`)
</ParamField>

<ParamField query="channel" type="string">
  Filter transactions by payment channel
  
  **Available channels:** `card`, `bank`, `ussd`, `qr`, `mobile_money`, `bank_transfer`
</ParamField>

<ParamField query="customer" type="string">
  Filter transactions by customer email or customer ID
</ParamField>

<ParamField query="amount" type="string">
  Filter transactions by exact amount (in smallest currency unit)
</ParamField>

<ParamField query="from" type="string">
  Start date for date range filter (ISO 8601 format: `YYYY-MM-DD` or `YYYY-MM-DDTHH:MM:SSZ`)
</ParamField>

<ParamField query="to" type="string">
  End date for date range filter (ISO 8601 format: `YYYY-MM-DD` or `YYYY-MM-DDTHH:MM:SSZ`)
</ParamField>

<ParamField query="sort" type="string" default="created_at">
  Field to sort by
  
  **Available fields:** `created_at`, `amount`, `status`, `paid_at`
</ParamField>

<ParamField query="order" type="string" default="desc">
  Sort order - `asc` (ascending) or `desc` (descending)
</ParamField>

## Response

<ResponseField name="status" type="string">
  Request status indicator - `"success"` or `"error"`
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable message describing the result
</ResponseField>

<ResponseField name="data" type="array">
  Array of transaction objects
  
  <Expandable title="Transaction Object">
    <ResponseField name="data[].id" type="integer">
      Internal transaction ID
    </ResponseField>
    
    <ResponseField name="data[].reference" type="string">
      Unique transaction reference
    </ResponseField>
    
    <ResponseField name="data[].amount" type="integer">
      Transaction amount in smallest currency unit
    </ResponseField>
    
    <ResponseField name="data[].currency" type="string">
      Transaction currency code
    </ResponseField>
    
    <ResponseField name="data[].status" type="string">
      Transaction status - `success`, `failed`, `pending`, `abandoned`, `cancelled`
    </ResponseField>
    
    <ResponseField name="data[].gateway_response" type="string">
      Response message from the payment gateway
    </ResponseField>
    
    <ResponseField name="data[].paid_at" type="string">
      ISO 8601 timestamp when payment was completed (null if not paid)
    </ResponseField>
    
    <ResponseField name="data[].created_at" type="string">
      ISO 8601 timestamp when transaction was created
    </ResponseField>
    
    <ResponseField name="data[].channel" type="string">
      Payment method used - `card`, `bank`, `ussd`, `qr`, `mobile_money`, `bank_transfer`
    </ResponseField>
    
    <ResponseField name="data[].fees" type="integer">
      Transaction fees charged in smallest currency unit
    </ResponseField>
    
    <ResponseField name="data[].customer" type="object">
      Customer information object
    </ResponseField>
    
    <ResponseField name="data[].authorization" type="object">
      Payment authorization details (for successful card payments)
    </ResponseField>
    
    <ResponseField name="data[].metadata" type="object">
      Custom metadata attached to the transaction
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata
  
  <Expandable title="Meta Object">
    <ResponseField name="meta.total" type="integer">
      Total number of transactions matching the filter criteria
    </ResponseField>
    
    <ResponseField name="meta.page" type="integer">
      Current page number
    </ResponseField>
    
    <ResponseField name="meta.per_page" type="integer">
      Number of transactions per page
    </ResponseField>
    
    <ResponseField name="meta.pages" type="integer">
      Total number of pages
    </ResponseField>
    
    <ResponseField name="meta.has_more" type="boolean">
      Whether there are more pages available
    </ResponseField>
  </Expandable>
</ResponseField>

## Pagination

PayVessel uses page-based pagination for the transactions list endpoint:

<CardGroup cols={2}>
  <Card title="Navigate Pages" icon="arrows-left-right">
    Use `page` parameter to navigate through pages:
    - `?page=1` - First page
    - `?page=2` - Second page
    - etc.
  </Card>
  
  <Card title="Control Page Size" icon="list">
    Use `per_page` parameter to control results per page:
    - Default: 50 transactions
    - Maximum: 100 transactions
  </Card>
</CardGroup>

### Pagination Example

```javascript
// Function to get all transactions
const getAllTransactions = async () => {
  let allTransactions = [];
  let page = 1;
  let hasMore = true;
  
  while (hasMore) {
    const response = await payvessel.transaction.list({
      page: page,
      per_page: 100 // Maximum per page
    });
    
    allTransactions = allTransactions.concat(response.data);
    hasMore = response.meta.has_more;
    page++;
  }
  
  return allTransactions;
};
```

## Common Use Cases

<Tabs>
  <Tab title="Dashboard Analytics">
    ```javascript
    // Get successful transactions for revenue calculation
    const getRevenueData = async (startDate, endDate) => {
      const transactions = await payvessel.transaction.list({
        status: 'success',
        from: startDate,
        to: endDate,
        per_page: 100
      });
      
      const revenue = transactions.data.reduce((total, txn) => {
        return total + (txn.amount - txn.fees);
      }, 0);
      
      return {
        total_revenue: revenue,
        transaction_count: transactions.meta.total,
        average_transaction: revenue / transactions.meta.total
      };
    };
    ```
  </Tab>
  
  <Tab title="Customer Statement">
    ```javascript
    // Get transaction history for a specific customer
    const getCustomerStatement = async (customerEmail, months = 3) => {
      const threeMonthsAgo = new Date();
      threeMonthsAgo.setMonth(threeMonthsAgo.getMonth() - months);
      
      const transactions = await payvessel.transaction.list({
        customer: customerEmail,
        from: threeMonthsAgo.toISOString().split('T')[0],
        sort: 'created_at',
        order: 'desc'
      });
      
      return transactions.data;
    };
    ```
  </Tab>
  
  <Tab title="Failed Payment Analysis">
    ```javascript
    // Analyze failed payments to improve success rates
    const analyzeFailedPayments = async () => {
      const failedTransactions = await payvessel.transaction.list({
        status: 'failed',
        per_page: 100
      });
      
      const failureReasons = {};
      failedTransactions.data.forEach(txn => {
        const reason = txn.gateway_response;
        failureReasons[reason] = (failureReasons[reason] || 0) + 1;
      });
      
      return failureReasons;
    };
    ```
  </Tab>
  
  <Tab title="Reconciliation">
    ```javascript
    // Daily reconciliation of successful payments
    const dailyReconciliation = async (date) => {
      const startOfDay = `${date}T00:00:00Z`;
      const endOfDay = `${date}T23:59:59Z`;
      
      const transactions = await payvessel.transaction.list({
        status: 'success',
        from: startOfDay,
        to: endOfDay,
        per_page: 100
      });
      
      return {
        date: date,
        transaction_count: transactions.meta.total,
        total_amount: transactions.data.reduce((sum, txn) => sum + txn.amount, 0),
        total_fees: transactions.data.reduce((sum, txn) => sum + txn.fees, 0)
      };
    };
    ```
  </Tab>

## Best Practices

<CardGroup cols={2}>
  <Card title="Efficient Pagination" icon="zap">
    Use appropriate `per_page` values. Start with 50, increase to 100 only when needed to reduce response times.
  </Card>
  
  <Card title="Date Range Limits" icon="calendar-days">
    Limit date ranges for better performance. Query large date ranges in smaller chunks for faster responses.
  </Card>
  
  <Card title="Filter Early" icon="filter">
    Apply filters to reduce the dataset before pagination. This improves performance and reduces bandwidth.
  </Card>
  
  <Card title="Cache Results" icon="database">
    Cache frequently accessed transaction lists (like daily summaries) to improve application performance.
  </Card>
</CardGroup>
